Процедура УстановитьДоступностьИТипСубконто()

	КоличествоСубконто = Объект.КорСчет.ВидыСубконто.Количество();
	
	Для Инд = 1 По 2 Цикл
		
		Если Инд <= КоличествоСубконто Тогда
			
			Элементы["КорСубконто" + Инд].Доступность = Истина;
			Элементы["КорСубконто" + Инд].ВыбиратьТип = Ложь;
			
			ТипЗначения = Объект.КорСчет.ВидыСубконто[Инд - 1].ВидСубконто.ТипЗначения;
			
			Объект["КорСубконто" + Инд] = ТипЗначения.ПривестиЗначение(Объект["КорСубконто" + Инд]);
			
			
		
		Иначе	
			
			Элементы["КорСубконто" + Инд].Доступность = Ложь;
			
			Объект["КорСубконто" + Инд] = Неопределено;
			
		КонецЕсли; 
		
	
	КонецЦикла;
	

КонецПроцедуры // УстановитьДоступностьИТипСубконто()
